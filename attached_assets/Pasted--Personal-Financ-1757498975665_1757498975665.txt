# --- 3. Helper Functions for Data Processing ---

def get_user_transactions(uid):
    """Fetches all transactions for a specific user ID."""
    if db is None:
        # Generate realistic mock data for development
        mock_data = []
        for i in range(24): # 24 months of data for better AI training
            date = datetime.now() - pd.DateOffset(months=i)
            # Simulate spending that matches your historical data
            amount = np.random.randint(3000, 7000) + (i * 50) 
            mock_data.append({"amount": amount, "date": date})
        return mock_data
    
    transactions_ref = db.collection('users').document(uid).collection('transactions').stream()
    data = []
    for doc in transactions_ref:
        entry = doc.to_dict()
        if 'amount' in entry and 'date' in entry and isinstance(entry['date'], datetime):
            data.append({"amount": entry['amount'], "date": entry['date']})
    return data

def prepare_monthly_data(data):
    """Aggregates raw transactions into a sequence of MONTHLY TOTALS for time-series forecasting."""
    if not data: return None
    df = pd.DataFrame(data)
    df['date'] = pd.to_datetime(df['date'])
    df.set_index('date', inplace=True)
    # This correctly resamples the data into monthly sums
    return df['amount'].resample('ME').sum()

# --- 4. Advanced and Robust Prediction Functions ---

def set_seeds(seed=42):
    """Ensures reproducibility for the AI model."""
    np.random.seed(seed)
    random.seed(seed)
    tf.random.set_seed(seed)

def create_supervised_series(series, n_steps):
    """Converts a time series into a supervised learning dataset for LSTMs."""
    values = series.values.reshape(-1, 1)
    scaler = MinMaxScaler(feature_range=(0, 1))
    scaled = scaler.fit_transform(values)
    X, y = [], []
    for i in range(len(scaled) - n_steps):
        X.append(scaled[i:i + n_steps, 0])
        y.append(scaled[i + n_steps, 0])
    X = np.array(X).reshape((-1, n_steps, 1))
    y = np.array(y)
    return X, y, scaler

def get_arima_prediction(monthly_data):
    """(Fallback Model) A reliable statistical model for users with less data."""
    if monthly_data is None or len(monthly_data) < 2: return None
    try:
        model = ARIMA(monthly_data, order=(1, 1, 0))
        model_fit = model.fit()
        return float(model_fit.forecast(steps=1).iloc[0])
    except Exception: return None

def get_lstm_prediction_improved(monthly_data, n_steps=6, epochs=200, batch_size=8, min_train_size=24):
    """(Primary Model) A powerful neural network that learns complex patterns from user data."""
    try:
        if monthly_data is None or len(monthly_data) < n_steps + 1: return None

        use_epochs = max(30, min(epochs, 200))
        if len(monthly_data) < min_train_size: use_epochs = 60

        set_seeds(1234)
        X, y, scaler = create_supervised_series(monthly_data, n_steps)
        split_idx = int(len(X) * 0.8)
        X_train, X_val = X[:split_idx], X[split_idx:]
        y_train, y_val = y[:split_idx], y[split_idx:]

        model = Sequential([
            Input(shape=(n_steps, 1)),
            LSTM(128, return_sequences=True),
            Dropout(0.25),
            LSTM(64, return_sequences=False),
            Dropout(0.2),
            Dense(32, activation='relu', kernel_regularizer=l2(1e-4)),
            Dense(1)
        ])
        model.compile(optimizer=Adam(learning_rate=1e-3), loss='mse')

        es = EarlyStopping(monitor='val_loss', patience=20, restore_best_weights=True, verbose=0)
        rlrop = ReduceLROnPlateau(monitor='val_loss', factor=0.5, patience=8, verbose=0, min_lr=1e-6)

        model.fit(X_train, y_train,
                  validation_data=(X_val, y_val) if len(X_val) > 0 else None,
                  epochs=use_epochs, batch_size=batch_size, verbose=0, callbacks=[es, rlrop])

        last_seq = monthly_data.values[-n_steps:].reshape(-1, 1)
        scaled_last = scaler.transform(last_seq).reshape(1, n_steps, 1)
        scaled_pred = model.predict(scaled_last, verbose=0)
        pred = scaler.inverse_transform(scaled_pred)[0, 0]

        tf.keras.backend.clear_session()
        return float(pred)

    except Exception as ex:
        print(f"⚠️ LSTM Improved failed: {ex}")
        tf.keras.backend.clear_session()
        return None

def get_ensemble_prediction(monthly_data):
    """The 'Team of Experts' model: Combines AI with statistical models and safety checks."""
    if monthly_data is None or len(monthly_data) == 0: return None

    lstm_pred = None
    if len(monthly_data) >= 18:
        lstm_pred = get_lstm_prediction_improved(monthly_data, n_steps=6)
    elif len(monthly_data) >= 9:
        lstm_pred = get_lstm_prediction_improved(monthly_data, n_steps=3, epochs=80)

    arima_pred = get_arima_prediction(monthly_data)
    recent_avg = float(monthly_data.tail(3).mean()) if len(monthly_data) >= 3 else float(monthly_data.mean())

    weights = {}
    if lstm_pred is not None and arima_pred is not None: weights = {'lstm': 0.7, 'arima': 0.25, 'avg': 0.05}
    elif lstm_pred is not None: weights = {'lstm': 0.9, 'avg': 0.1}
    elif arima_pred is not None: weights = {'arima': 0.9, 'avg': 0.1}
    else: weights = {'avg': 1.0}

    final, total_weight = 0.0, 0.0
    if 'lstm' in weights and lstm_pred is not None:
        final += weights['lstm'] * lstm_pred
        total_weight += weights['lstm']
    if 'arima' in weights and arima_pred is not None:
        final += weights['arima'] * arima_pred
        total_weight += weights['arima']
    if 'avg' in weights:
        final += weights['avg'] * recent_avg
        total_weight += weights['avg']

    if total_weight == 0: return recent_avg
    ensemble_pred = final / total_weight

    # Final safety net to ensure the prediction is always realistic
    if ensemble_pred < (recent_avg * 0.5):
        ensemble_pred = recent_avg
    if ensemble_pred > (recent_avg * 5):
        ensemble_pred = recent_avg * 2.0

    return float(ensemble_pred)

@app.route("/predict", methods=["POST"])
def predict():
    # Authentication logic to protect the endpoint
    uid = None
    if db is None:
        uid = "dev_user_123"
    else:
        auth_header = request.headers.get('Authorization')
        if not auth_header or not auth_header.startswith('Bearer '): return jsonify({"error": "Unauthorized"}), 401
        try:
            id_token = auth_header.split('Bearer ')[1]
            decoded_token = auth.verify_id_token(id_token)
            uid = decoded_token['uid']
        except Exception as e: return jsonify({"error": f"Unauthorized: {str(e)}"}), 401
            
    try:
        transactions = get_user_transactions(uid)
        monthly_series = prepare_monthly_data(transactions)
        
        # Use the powerful ensemble model to get a single, reliable prediction
        final_prediction = get_ensemble_prediction(monthly_series)

        if db is not None:
            prediction_ref = db.collection('users').document(uid).collection('prediction').document('next_month')
            prediction_ref.set({
                'predicted_expense': round(final_prediction, 2) if final_prediction is not None else None,
                'updatedAt': firestore.SERVER_TIMESTAMP
            })
            print(f"💾 Prediction saved for user {uid}: {final_prediction}")

        return jsonify({"message": "Prediction successful", "predicted_expense": final_prediction}), 200

    except Exception as e:
        print(f"🔥 Internal server error for user {uid}: {e}")
        return jsonify({"error": "An internal server error occurred."}), 500
